<script>
  // Particles
  const container = document.getElementById('particles');
  for (let i = 0; i < 20; i++) {
    const p = document.createElement('div');
    p.className = 'particle';
    p.style.top = (i * 5 + 5) + '%';
    p.style.left = (i * 5 + 2) + '%';
    p.style.animationDelay = (-i * 0.4) + 's';
    p.style.animationDuration = (7 + Math.random() * 4) + 's';
    container.appendChild(p);
  }

  // SoundCloud Widget
  const PLAY_ICON  = '<path d="M8 5v14l11-7z"/>';
  const PAUSE_ICON = '<path d="M6 19h4V5H6v14zm8-14v14h4V5h-4z"/>';

  function fmt(ms) {
    const s = Math.floor(ms / 1000), m = Math.floor(s / 60);
    return m + ':' + String(s % 60).padStart(2, '0');
  }

  const playBtn = document.getElementById('playBtn');
  const disc    = document.getElementById('disc');
  const hint    = document.getElementById('hint');
  const seekBar = document.getElementById('seek');
  const posEl   = document.getElementById('pos');
  const durEl   = document.getElementById('dur');
  const volBar  = document.getElementById('vol');
  let playing = false, duration = 0;

  const widget = SC.Widget(document.getElementById('sc-player'));

  widget.bind(SC.Widget.Events.READY, () => {
    playBtn.disabled = false;
    hint.style.display = 'none';
    widget.setVolume(80);
  });

  widget.bind(SC.Widget.Events.PLAY, () => {
    playing = true;
    playBtn.querySelector('svg').innerHTML = PAUSE_ICON;
    disc.classList.add('spinning');
  });

  widget.bind(SC.Widget.Events.PAUSE, () => {
    playing = false;
    playBtn.querySelector('svg').innerHTML = PLAY_ICON;
    disc.classList.remove('spinning');
  });

  widget.bind(SC.Widget.Events.FINISH, () => {
    playing = false;
    playBtn.querySelector('svg').innerHTML = PLAY_ICON;
    disc.classList.remove('spinning');
  });

  widget.bind(SC.Widget.Events.PLAY_PROGRESS, (e) => {
    posEl.textContent = fmt(e.currentPosition);
    if (e.relativePosition > 0) {
      duration = e.currentPosition / e.relativePosition;
      seekBar.max = Math.floor(duration);
      durEl.textContent = fmt(duration);
    }
    seekBar.value = Math.floor(e.currentPosition);
  });

  playBtn.addEventListener('click', () => {
    playing ? widget.pause() : widget.play();
  });

  seekBar.addEventListener('input', () => {
    widget.seekTo(Number(seekBar.value));
  });

  volBar.addEventListener('input', () => {
    widget.setVolume(Number(volBar.value));
  });
</script>
